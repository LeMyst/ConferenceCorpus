[docs]    def getListOfDicts(self):
        '''
        get my list of dicts
        '''
        wikiCFP=datasources.wikicfpscrape.WikiCfpScrape()
        if not wikiCFP.em.isCached():
            wikiCFP.cacheEvents()
        else:
            wikiCFP.em.fromStore()
        lod=[]
        for event in wikiCFP.em.events:
            lod.append(event.__dict__)
        return lod
